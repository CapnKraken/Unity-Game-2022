// include the standard library
INCLUDE stdlib.txt

Set direction = 0

Repeat 5
	
	Set xPos = 0
	Set speed = 1
	Set burst = 5
	Set whip = 1
	
	Wait 0.5

	Repeat 5
		
		Shoot
		Set direction + 10
		Set speed + 1.5

		Wait 0.2

	EndRepeat

	Set whipIncrement = 1
	Set speed = 1
	Set direction = -90
	
	Repeat 3
		Set burst = 1
		Set whip = 4
		Shoot

		Set whipIncrement + 1
		Wait 1

	EndRepeat
EndRepeat
