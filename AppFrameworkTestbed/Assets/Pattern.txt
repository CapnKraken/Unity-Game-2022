// include the standard library
INCLUDE stdlib.txt

// double border of wave and particle pattern

// variables
set p_leftdirection = 0
set p_rightdirection = 0
set p_rotationspeed = 0

Define BorderWaveParticle
ResetVars

	set p_leftdirection = 90
	set p_rightdirection = 90
	set p_rotationspeed = 0

	set speed = 6
	set burst = 2

	Repeat 200

		set direction = p_leftdirection
		set xPos = 2
		Shoot

		set direction = p_rightdirection
		set xPos = -2
		Shoot

		set p_leftdirection + p_rotationspeed
		set p_rightdirection - p_rotationspeed
		set p_rotationspeed + 0.7
		Wait 0.05

	EndRepeat
EndFunction

Repeat 1000
	
	// BorderWaveParticle

	ResetVars

	Repeat 100

		AngleToPlayer direction location 0 0
		Set burst = 1
		Set speed = 4
		Set spread = 15
		Set spreadDistance = 5

		Shoot

		Wait 1

	EndRepeat

	ResetVars

	Repeat 5
		Random direction 0.0 360.0
		Set burst = 15
		Random speed 2.1 5.5

		Shoot

		Wait 1
	EndRepeat

EndRepeat




IGNORE

Repeat 5
	
	Set xPos = 0
	Set speed = 1
	Set burst = 5
	Set whip = 1
	
	Wait 0.5

	Set min = 0.0
	Random direction min 360.0


	Repeat 5
		
		Shoot
		Set direction + 10
		Set speed + 1.5

		Wait 0.2

	EndRepeat

	Set whipIncrement = 1
	Set speed = 1
	Set direction = -90
	
	Repeat 3
		Set burst = 1
		Set whip = 4
		Shoot

		Set whipIncrement + 1
		Wait 1

	EndRepeat
EndRepeat
ENDIGNORE
